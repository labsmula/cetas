let UID=0;
function mkUnit(def, stars=1, enemy=false){
  const m = stars===2?1.8:stars===3?3.2:1;
  return {
    ...def, uid:++UID, stars, enemy,
    maxHp:Math.round(def.hp*m), curHp:Math.round(def.hp*m),
    atkVal:Math.round(def.atk*m),
    x:0, y:0, vx:0, vy:0, anim:0, animDir:1, shake:0,
    attackTimer:0, dead:false, floats:[]
  };
}

let G = initGame();

    round:1, hp:100, gold:10, phase:'prep',
    maxBoardSlots:3,
    board: Array(ROWS).fill(null).map(()=>Array(COLS).fill(null)),
    bench: Array(8).fill(null),
    shop:[],
    selected:null,
    animFrame:0, battleRunning:false,

}

  for(let r=2;r<ROWS;r++) for(let c=0;c<COLS;c++) if(G.board[r][c]&&!G.board[r][c].enemy) n++;
  return n;
}

function checkMerge(){
  const counts={};
  G.bench.forEach((u,i)=>{ if(!u) return; const k=u.id+'_'+u.stars; if(!counts[k]) counts[k]={idxs:[],def:u}; counts[k].idxs.push(i); });
  for(const k in counts){
    const {idxs,def}=counts[k];
    if(idxs.length>=3&&def.stars<3){
      const newU=mkUnit(UNIT_DEFS.find(d=>d.id===def.id),def.stars+1);
      idxs.slice(0,3).forEach(i=>G.bench[i]=null);
      const free=G.bench.indexOf(null);
      if(free>=0) G.bench[free]=newU; else G.bench[idxs[0]]=newU;
      log(`✨ Merge! ${def.name} ⭐${def.stars+1} muncul!`);
      checkMerge(); return;
    }
  }
}

function drawPixelChar(ctx, def, stars, x, y, size, flip, anim, shake){
  const body=def.body;
  const pw=Math.floor(size/4);
  const rows=body.length;
  const sx=flip?x+size-pw:x;
  const dx=flip?-pw:pw;
  const shakeX=shake?(Math.random()*4-2)|0:0;
  const shakeY=shake?(Math.random()*4-2)|0:0;
  const bobY=Math.sin(anim*0.12)*2;

  body.forEach((row,ri)=>{
    row.forEach((px,ci)=>{
      if(!px) return;
      const px_x=(flip?x+size-(ci+1)*pw:x+ci*pw)+shakeX;
      const px_y=y+ri*pw+bobY+shakeY;
      ctx.fillStyle=(ri===0&&(ci===1||ci===2))?def.accent:def.color;
      ctx.fillRect(px_x, px_y, pw, pw);
    });
  });
  if(stars>1){
    ctx.fillStyle='#EF9F27';
    ctx.font=`bold ${Math.max(8,pw)}px sans-serif`;
    ctx.fillText('★'.repeat(stars), x, y-2);
  }
}

function drawHpBar(ctx, u, bx, by, bw){
  const pct=u.curHp/u.maxHp;
  ctx.fillStyle='#333';
  ctx.fillRect(bx,by,bw,3);
  ctx.fillStyle=pct>0.5?'#5DCAA5':pct>0.25?'#EF9F27':'#E24B4A';
  ctx.fillRect(bx,by,Math.round(bw*pct),3);
}

function getUnitPos(r,c){

function drawBoard(){
  CX.clearRect(0,0,660,260);
  for(let r=0;r<ROWS;r++){
    for(let c=0;c<COLS;c++){
      const isEnemy=r<2;
      const isSelected=G.selected&&G.selected.src==='board'&&G.selected.r===r&&G.selected.c===c;
      CX.fillStyle=isEnemy?'#2a1a1a':(r%2+c%2)%2===0?'#1e2535':'#232a3d';
      CX.fillRect(c*CW,r*CH,CW,CH);
      if(isSelected){ CX.fillStyle='rgba(29,158,117,0.35)'; CX.fillRect(c*CW,r*CH,CW,CH); }
      if(G.phase==='prep'&&!isEnemy&&!isSelected&&G.selected&&getBoardUnitCount()<G.maxBoardSlots){
        if(!G.board[r][c]){ CX.fillStyle='rgba(29,158,117,0.12)'; CX.fillRect(c*CW,r*CH,CW,CH); }
      }
      CX.strokeStyle='rgba(255,255,255,0.04)';
      CX.strokeRect(c*CW,r*CH,CW,CH);
      const u=G.board[r][c];
      if(u&&!u.dead){
        const ux=c*CW+2, uy=r*CH+4, sz=Math.min(CW-4,CH-10);
        drawPixelChar(CX,UNIT_DEFS.find(d=>d.id===u.id),u.stars,ux,uy,sz,!u.enemy,u.anim,u.shake>0);
        drawHpBar(CX,u,c*CW+2,r*CH+CH-5,CW-4);
        if(u.shake>0) u.shake--;
        u.floats=u.floats.filter(f=>{
          CX.globalAlpha=f.life/20;
          CX.fillStyle=f.color||'#E24B4A';
          CX.font='bold 11px sans-serif';
          CX.fillText(f.txt, c*CW+CW/2-8, r*CH+CH/2-f.rise);
          f.rise+=1.2; f.life--;
          CX.globalAlpha=1;
          return f.life>0;
        });
      }

        CX.globalAlpha=0.25;
        CX.fillStyle='#555';
        CX.fillRect(c*CW+4,r*CH+4,CW-8,CH-8);
        CX.globalAlpha=1;
      }
    }
  }
  CX.fillStyle='rgba(255,255,255,0.08)';

  CX.fillStyle='rgba(255,255,255,0.06)';

CV.addEventListener('click',e=>{
  if(G.phase!=='prep') return;
  const rect=CV.getBoundingClientRect();
  const mx=e.clientX-rect.left, my=e.clientY-rect.top;
  const c=Math.floor(mx/CW), r=Math.floor(my/CH);
  if(r<0||r>=ROWS||c<0||c>=COLS) return;
  const isEnemy=r<2;
  if(isEnemy){ if(G.selected) G.selected=null; updateUI(); return; }

  if(G.selected){
    const sel=G.selected;
    let unit=null;
    if(sel.src==='bench'){ unit=G.bench[sel.idx]; G.bench[sel.idx]=null; }
    else { unit=G.board[sel.r][sel.c]; G.board[sel.r][sel.c]=null; }
    const existing=G.board[r][c];
    if(existing){
      if(getBoardUnitCount()>=G.maxBoardSlots&&!existing.enemy){
        if(sel.src==='bench') G.bench[sel.idx]=unit;
        else G.board[sel.r][sel.c]=unit;
        G.selected=null;
        log(`⚠️ Slot penuh! Maksimal ${G.maxBoardSlots} unit di papan.`);
        updateUI(); return;
      }
      if(sel.src==='bench') G.bench[sel.idx]=existing;
      else G.board[sel.r][sel.c]=existing;
    } else {
      const tempCount=getBoardUnitCount()+(sel.src==='bench'?1:0);
      if(sel.src==='bench'&&tempCount>G.maxBoardSlots){
        G.bench[sel.idx]=unit;
        G.selected=null;
        log(`⚠️ Slot penuh! Maksimal ${G.maxBoardSlots} unit di papan.`);
        updateUI(); return;
      }
    }
    G.board[r][c]=unit;
    G.selected=null;
    log(`📍 ${unit.name} ditempatkan`);
  } else {
    if(G.board[r][c]&&!G.board[r][c].enemy) G.selected={src:'board',r,c};
  }
  updateUI();
});

function onBenchClick(i){
  if(G.phase!=='prep') return;
  if(G.selected){
    const sel=G.selected;
    let unit=null;
    if(sel.src==='bench'){ unit=G.bench[sel.idx]; G.bench[sel.idx]=null; }
    else { unit=G.board[sel.r][sel.c]; G.board[sel.r][sel.c]=null; }
    const existing=G.bench[i];

    if(existing){
      if(sel.src==='board') G.board[sel.r][sel.c]=existing;
      else G.bench[sel.idx]=existing;
    }
    G.selected=null;
  } else {
    if(G.bench[i]) G.selected={src:'bench',idx:i};
  }
  updateUI();
}

function buyUnit(i){
  const u=G.shop[i];
  if(u.sold) return;
  if(G.gold<u.cost){ log('🪙 Koin tidak cukup!'); return; }
  const free=G.bench.indexOf(null);
  if(free<0){ log('⚠️ Bangku penuh! Jual unit dulu.'); return; }
  G.gold-=u.cost;
  G.shop[i].sold=true;
  G.bench[free]=mkUnit(u);
  log(`🛒 Beli ${u.name} −🪙${u.cost}`);
  checkMerge();
  updateUI();
}

function genEnemies(){
  for(let r=0;r<2;r++) for(let c=0;c<COLS;c++) G.board[r][c]=null;
  const maxEnemy=Math.min(G.maxBoardSlots+1, 6);
  const count=Math.min(2+G.round, maxEnemy);
  const tier=Math.min(Math.floor(G.round/2),2);
  const pool=UNIT_DEFS.filter(u=>u.cost<=1+tier);
  for(let i=0;i<count;i++){
    const def=pool[Math.floor(Math.random()*pool.length)];
    const stars=G.round>=4&&Math.random()<0.35?2:1;
    let placed=false;
    for(let attempt=0;attempt<20&&!placed;attempt++){
      const r=Math.floor(Math.random()*2), c=Math.floor(Math.random()*COLS);
      if(!G.board[r][c]){ G.board[r][c]=mkUnit(def,stars,true); placed=true; }
    }
  }
}

function getAllUnits(){
  const all=[];
  for(let r=0;r<ROWS;r++) for(let c=0;c<COLS;c++) if(G.board[r][c]) all.push({u:G.board[r][c],r,c});
  return all;
}

function runBattleStep(){
  const all=getAllUnits();
  const alive=all.filter(x=>!x.u.dead);
  const allies=alive.filter(x=>!x.u.enemy);
  const enemies=alive.filter(x=>x.u.enemy);
  if(!allies.length||!enemies.length) return false;

  alive.forEach(x=>{ x.u.anim+=x.u.animDir; if(x.u.anim>20||x.u.anim<0) x.u.animDir*=-1; });

  alive.forEach(x=>{
    x.u.attackTimer+=x.u.spd;
    if(x.u.attackTimer<1) return;
    x.u.attackTimer-=1;
    const targets=x.u.enemy?allies:enemies;
    if(!targets.length) return;
    const t=targets[Math.floor(Math.random()*targets.length)];
    const dmg=Math.max(1,x.u.atkVal+((Math.random()*8|0)-4));
    t.u.curHp=Math.max(0,t.u.curHp-dmg);
    t.u.shake=4;
    t.u.floats.push({txt:'-'+dmg,rise:0,life:20,color:t.u.enemy?'#F0997B':'#E24B4A'});
    if(t.u.curHp<=0) t.u.dead=true;
  });
  return true;
}

  genEnemies();
  updateUI();
  log('⚔️ Pertempuran dimulai!');

  let ticks=0;
  battleInterval=setInterval(()=>{
    ticks++;
    const cont=runBattleStep();
    if(!cont||ticks>300){
      clearInterval(battleInterval);
      endBattle();
    }
  },60);
});

function endBattle(){
  const all=getAllUnits();
  const alliesAlive=all.filter(x=>!x.u.enemy&&!x.u.dead);
  const enemiesAlive=all.filter(x=>x.u.enemy&&!x.u.dead);
  const win=enemiesAlive.length===0;

  if(win){
    const earn=3+G.round;
    G.gold=Math.min(G.gold+earn,20);
    G.maxBoardSlots=Math.min(G.maxBoardSlots+1,7);
    showModal('🏆 Menang!','#1D9E75',`${alliesAlive.length} unit selamat! +🪙${earn} koin. Slot +1 (kini ${G.maxBoardSlots})`);
    log(`🏆 Ronde ${G.round} MENANG! +🪙${earn}, slot naik ke ${G.maxBoardSlots}`);
  } else {
    const dmg=10+enemiesAlive.length*6;
    G.hp=Math.max(0,G.hp-dmg);
    showModal('😤 Kalah!','#D85A30',`−${dmg} HP. Sisa HP: ${G.hp}`);
    log(`😤 Ronde ${G.round} KALAH! −${dmg} HP`);
  }
}

function showModal(title,color,desc){
  document.getElementById('m-title').textContent=title;
  document.getElementById('m-title').style.color=color;
  document.getElementById('m-desc').textContent=desc;
  const btn=document.getElementById('btn-next');
  if(G.hp<=0){ btn.textContent='Main Lagi'; document.getElementById('m-title').textContent='☠️ Game Over'; document.getElementById('m-desc').textContent='HP habis! Pertandingan selesai.'; }
  else if(G.round>=5){ btn.textContent='Main Lagi'; }
  else { btn.textContent=`Ronde ${G.round+1} →`; }
  document.getElementById('modal-wrap').classList.add('show');
}

document.getElementById('btn-next').addEventListener('click',()=>{
  document.getElementById('modal-wrap').classList.remove('show');
  if(G.hp<=0||G.round>=5){
    G=initGame(); genShop(); log('🔄 Permainan baru! Beli unit & susun formasi.'); updateUI(); return;
  }
  G.round++;
  G.phase='prep';
  G.gold=Math.min(G.gold+5,20);
  for(let r=0;r<2;r++) for(let c=0;c<COLS;c++) G.board[r][c]=null;
  for(let r=2;r<ROWS;r++) for(let c=0;c<COLS;c++){
    if(G.board[r][c]&&!G.board[r][c].dead){
      const u=G.board[r][c];
      u.curHp=Math.min(u.maxHp,u.curHp+Math.floor(u.maxHp*0.25));
      u.anim=0; u.attackTimer=0; u.floats=[];
    } else if(G.board[r][c]&&G.board[r][c].dead){
      G.board[r][c]=null;
    }
  }
  genShop();
  log(`📋 Ronde ${G.round}. +5🪙 Toko diperbarui. Slot: ${G.maxBoardSlots}`);
  updateUI();
});

function loop(){
  drawBoard();
  requestAnimationFrame(loop);
}

genShop(); updateUI(); loop();
</script>
